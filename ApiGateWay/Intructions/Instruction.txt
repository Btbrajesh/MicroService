Login Api Route and All Api will redirect from one port 

// Login Api

curl --location 'http://localhost:4433/gateway/account' \
--header 'Content-Type: application/json' \
--data '{
    "UserName": "admin",
    "Password": "admin123"  
}'

curl --location 'http://localhost:4433/gateway/account' \
--header 'Content-Type: application/json' \
--data '{
    "UserName": "user01",
    "Password": "user01"  
}'
